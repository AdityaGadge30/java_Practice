
public class Main
{
    public static String concat_string(String s1 , String s2){
        
       return s1.concat(s2);
        
    }
    
	public static void main(String[] args) {
	    
	    String s1 = "hello";
	    String s2 = "hii";
	    
	    
	    System.out.print(concat_string(s1,s2));
	    
	}
}
