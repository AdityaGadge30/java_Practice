
public class Main
{
    public static void reverse_string(String s1){
        
    
        
        for(int i=s1.length()-1;i>=0;i--){
            
         System.out.print(s1.charAt(i));
           
        }
        
        
    }
    
	public static void main(String[] args) {
	    
	    String s1 = "hello kjbjkbk";
	    
	    
	    reverse_string(s1);
	    
	}
}
